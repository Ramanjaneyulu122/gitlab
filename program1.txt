public class Addition {
    public static void main(String[] args) {
        // Define the two numbers to be added
        int number1 = 5;
        int number2 = 10;
        
        // Perform the addition
        int sum = number1 + number2;
        
        // Display the result
        System.out.println("The sum of " + number1 + " and " + number2 + " is: " + sum);
    }
}
